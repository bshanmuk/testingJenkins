Installing and setting Java and Jenkins on GCE $ ssh username@IPAddressOfGCE $ sudo yum install java-1.8.0-openjdk $ sudo wget -O /etc/yum.repos.d/jenkins.repo http://pkg.jenkins-ci.org/redhat-stable/jenkins.repo $ sudo rpm --import https://jenkins-ci.org/redhat/jenkins-ci.org.key $ sudo yum install jenkins $ sudo systemctl start jenkins $ sudo /sbin/chkconfig jenkins on $ sudo firewall-cmd --zone=public --add-port=8080/tcp --permanent $ sudo firewall-cmd --reload
